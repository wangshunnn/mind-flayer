use serde::{Deserialize, Serialize};
use std::sync::{Arc, RwLock};
use tauri::Manager;

#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct TelegramConfig {
    pub enabled: bool,
    pub allowed_user_ids: Vec<String>,
}

#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct ChannelRuntimeConfig {
    pub selected_model: Option<String>,
    pub telegram: TelegramConfig,
    /// Skill IDs disabled for this channel session (runtime-only, not persisted)
    pub disabled_skills: Vec<String>,
}

/// Thread-safe wrapper for runtime channel config.
#[derive(Clone)]
pub struct ChannelRuntimeConfigState {
    inner: Arc<RwLock<ChannelRuntimeConfig>>,
}

impl ChannelRuntimeConfigState {
    pub fn new() -> Self {
        Self {
            inner: Arc::new(RwLock::new(ChannelRuntimeConfig::default())),
        }
    }

    pub fn get(&self) -> ChannelRuntimeConfig {
        self.inner.read().unwrap().clone()
    }

    pub fn update(&self, updater: impl FnOnce(&mut ChannelRuntimeConfig)) {
        let mut guard = self.inner.write().unwrap();
        updater(&mut guard);
    }
}

impl Default for ChannelRuntimeConfigState {
    fn default() -> Self { Self::new() }
}

/// Called by frontend settings page when channel config changes.
#[tauri::command]
pub async fn update_channel_runtime_config(
    app: tauri::AppHandle,
    selected_model: Option<String>,
    telegram: Option<TelegramConfig>,
    disabled_skills: Option<Vec<String>>,
) -> Result<(), String> {
    let state = app.state::<ChannelRuntimeConfigState>().inner().clone();
    state.update(|cfg| {
        if let Some(m) = selected_model {
            cfg.selected_model = Some(m);
        }
        if let Some(t) = telegram {
            cfg.telegram = t;
        }
        if let Some(s) = disabled_skills {
            cfg.disabled_skills = s;
        }
    });
    Ok(())
}

// src-tauri/src/telegram/mod.rs
use std::sync::{Arc, Mutex};
use tokio_util::sync::CancellationToken;

/// Runtime state for the Telegram bot.
#[derive(Clone)]
pub struct TelegramBotService {
    abort: Arc<Mutex<Option<CancellationToken>>>,
}

impl TelegramBotService {
    pub fn new() -> Self {
        Self {
            abort: Arc::new(Mutex::new(None)),
        }
    }

    pub fn is_running(&self) -> bool {
        self.abort.lock().unwrap().is_some()
    }

    /// Start the bot with the given token. No-op if already running.
    pub async fn start(
        &self,
        token: String,
        allowed_user_ids: Vec<String>,
        app_handle: tauri::AppHandle,
    ) {
        let mut guard = self.abort.lock().unwrap();
        if guard.is_some() {
            return; // already running
        }
        let abort = CancellationToken::new();
        *guard = Some(abort.clone());
        drop(guard);

        let service = self.clone();
        tokio::spawn(async move {
            run_bot(token, allowed_user_ids, app_handle, abort.clone()).await;
            // Clear abort token when done
            let mut g = service.abort.lock().unwrap();
            *g = None;
        });
    }

    /// Stop the bot. No-op if not running.
    pub fn stop(&self) {
        let mut guard = self.abort.lock().unwrap();
        if let Some(token) = guard.take() {
            token.cancel();
        }
    }
}

impl Default for TelegramBotService {
    fn default() -> Self { Self::new() }
}

async fn run_bot(
    token: String,
    allowed_user_ids: Vec<String>,
    app_handle: tauri::AppHandle,
    abort: CancellationToken,
) {
    use teloxide::prelude::*;
    use teloxide::types::Message;

    log::info!("[Telegram] Bot starting");

    let bot = Bot::new(&token);

    // Verify the token is valid before entering dispatch loop
    match bot.get_me().await {
        Ok(me) => log::info!(
            "[Telegram] Bot connected: @{}",
            me.username.as_deref().unwrap_or("unknown")
        ),
        Err(e) => {
            log::error!("[Telegram] Failed to connect: {e}");
            return;
        }
    }

    let allowed_ids: Arc<Vec<String>> = Arc::new(allowed_user_ids);
    let app_handle = Arc::new(app_handle);

    let handler = dptree::entry().branch(
        Update::filter_message().endpoint({
            let allowed = Arc::clone(&allowed_ids);
            let app = Arc::clone(&app_handle);
            move |bot: Bot, msg: Message| {
                let allowed = Arc::clone(&allowed);
                let app = Arc::clone(&app);
                handle_message(bot, msg, allowed, app)
            }
        }),
    );

    let mut dispatcher = Dispatcher::builder(bot, handler)
        .enable_ctrlc_handler()
        .build();

    tokio::select! {
        _ = abort.cancelled() => {
            log::info!("[Telegram] Bot stopped by cancellation");
        }
        _ = dispatcher.dispatch() => {
            log::info!("[Telegram] Bot dispatcher ended");
        }
    }
}

async fn handle_message(
    bot: teloxide::Bot,
    msg: teloxide::types::Message,
    allowed_user_ids: Arc<Vec<String>>,
    app_handle: Arc<tauri::AppHandle>,
) -> Result<(), teloxide::RequestError> {
    use teloxide::prelude::*;

    let user_id = msg
        .from
        .as_ref()
        .map(|u| u.id.to_string())
        .unwrap_or_default();

    // Check whitelist
    if !allowed_user_ids.is_empty() && !allowed_user_ids.contains(&user_id) {
        bot.send_message(msg.chat.id, "Access denied. Contact the bot owner.")
            .await?;
        return Ok(());
    }

    let text = match msg.text() {
        Some(t) => t.to_string(),
        None => return Ok(()), // ignore non-text messages
    };

    // Run the agentic loop for this message
    // Use a synthetic window_label for telegram sessions: "telegram:{chat_id}"
    let window_label = format!("telegram:{}", msg.chat.id);
    let chat_id_str = format!("telegram-{}", msg.chat.id);

    // Get provider config from keychain
    use crate::keychain;
    let configs = keychain::get_all_configs_providers();
    let provider_name = "openai"; // TODO: read from ChannelRuntimeConfig
    let provider_config = configs.get(provider_name).cloned();

    let Some(provider_cfg) = provider_config else {
        bot.send_message(msg.chat.id, "No AI provider configured.").await?;
        return Ok(());
    };

    // Build a simple message history (one user message for now)
    let messages = vec![crate::ai::ChatMessage {
        id: uuid::Uuid::new_v4().to_string(),
        role: "user".into(),
        content: text,
        tool_calls: vec![],
        tool_results: vec![],
    }];

    // For Telegram, we use a home folder as the project context
    let folder = dirs::home_dir().unwrap_or_else(|| std::path::PathBuf::from("/"));
    let app_data = match app_handle.path().app_data_dir() {
        Ok(d) => d,
        Err(e) => {
            log::error!("[Telegram] Failed to get app data dir: {e}");
            return Ok(());
        }
    };

    let abort = CancellationToken::new();
    let pending_approvals = Arc::new(dashmap::DashMap::new());

    // Collect response text via a channel (we can't emit Tauri events for Telegram)
    let (resp_tx, mut resp_rx) = tokio::sync::mpsc::channel::<String>(64);

    let params = crate::ai::stream::AgentLoopParams {
        window_label: window_label.clone(),
        folder: folder.clone(),
        chat_id: chat_id_str,
        messages,
        model: "gpt-4o".to_string(), // TODO: read from ChannelRuntimeConfig
        provider_name: provider_name.to_string(),
        enabled_tools: vec![], // No tools for Telegram by default
        abort: abort.clone(),
        pending_approvals,
        app_handle: (*app_handle).clone(),
        app_data_dir: app_data,
    };

    // Run the agentic loop (it will emit Tauri events, which are ignored for Telegram)
    // We collect the response by reading stream:chunk events.
    // Simpler approach: run provider directly and collect text
    let provider = crate::ai::provider::create_provider(
        provider_name,
        &provider_cfg.api_key,
        provider_cfg.base_url.as_deref(),
    );

    use crate::ai::provider::StreamConfig;
    use crate::ai::{ProviderEvent, ProviderMessage, MsgRole, Part};
    use crate::skills::injector::build_system_prompt;
    use tauri::Manager;

    let skill_catalog = app_handle
        .try_state::<crate::skills::catalog::SkillCatalog>()
        .map(|c| c.enabled())
        .unwrap_or_default();
    let system = build_system_prompt(
        &folder.to_string_lossy(),
        &skill_catalog,
        "gpt-4o",
    );

    let provider_messages = vec![ProviderMessage {
        role: MsgRole::User,
        parts: vec![Part::Text(msg.text().unwrap_or("").to_string())],
    }];

    let (event_tx, mut event_rx) = tokio::sync::mpsc::channel(64);
    let abort2 = CancellationToken::new();
    tokio::spawn(async move {
        let _ = provider
            .stream_chat(
                provider_messages,
                vec![],
                StreamConfig {
                    model: "gpt-4o".to_string(),
                    system_prompt: system,
                    abort: abort2,
                },
                event_tx,
            )
            .await;
    });

    let mut response_text = String::new();
    while let Some(event) = event_rx.recv().await {
        match event {
            ProviderEvent::TextDelta { delta } => response_text.push_str(&delta),
            ProviderEvent::StopReason(_) | ProviderEvent::Error(_) => break,
            _ => {}
        }
    }

    if response_text.is_empty() {
        response_text = "Sorry, I couldn't generate a response.".to_string();
    }

    bot.send_message(msg.chat.id, response_text).await?;
    Ok(())
}

// src/hooks/use-session.ts
import type { UIMessage } from "ai"
import { invoke } from "@tauri-apps/api/core"
import { getCurrentWebviewWindow } from "@tauri-apps/api/webviewWindow"
import { nanoid } from "nanoid"
import { useCallback, useEffect, useRef, useState } from "react"

// ── IPC contract types (must match src-tauri/src/ai/mod.rs) ──────────────────

export interface ChatMessage {
  id: string
  role: string
  content: string
  tool_calls?: ToolCallPart[]
  tool_results?: ToolResultPart[]
}

interface ToolCallPart {
  id: string
  name: string
  args: unknown
}

interface ToolResultPart {
  tool_call_id: string
  content: string
  is_error: boolean
}

export interface ToolApprovalRequest {
  toolCallId: string
  toolName: string
  args: unknown
}

// ── Tauri event payloads ───────────────────────────────────────────────────────

interface StreamChunkPayload {
  delta: string
}

interface StreamToolCallPayload {
  tool_call_id: string
  tool_name: string
  args: unknown
}

interface StreamToolResultPayload {
  tool_call_id: string
  result: unknown
  is_error: boolean
}

interface StreamDonePayload {
  usage: { prompt_tokens: number; completion_tokens: number }
}

interface StreamErrorPayload {
  message: string
}

interface ToolApprovalRequestPayload {
  tool_call_id: string
  tool_name: string
  args: unknown
}

// ── useSession ────────────────────────────────────────────────────────────────

export interface UseSessionOptions {
  folder: string
  chatId: string | null
  model: string
  provider: string
  enabledTools?: string[]
  onToolApprovalRequest?: (req: ToolApprovalRequest) => void
}

export interface UseSessionReturn {
  messages: UIMessage[]
  setMessages: (msgs: UIMessage[]) => void
  isLoading: boolean
  error: Error | null
  clearError: () => void
  append: (text: string) => Promise<void>
  stop: () => void
  reload: () => Promise<void>
}

export function useSession(options: UseSessionOptions): UseSessionReturn {
  const windowLabel = getCurrentWebviewWindow().label

  const [messages, setMessages] = useState<UIMessage[]>([])
  const [isLoading, setIsLoading] = useState(false)
  const [error, setError] = useState<Error | null>(null)

  // Internal refs to avoid stale closures
  const messagesRef = useRef<UIMessage[]>([])
  const optionsRef = useRef(options)
  const lastUserMessageRef = useRef<ChatMessage | null>(null)

  // Keep ref in sync
  useEffect(() => {
    messagesRef.current = messages
    optionsRef.current = options
  })

  // ── Tauri event listeners ────────────────────────────────────────────────────
  useEffect(() => {
    if (!options.chatId) return

    const unlisten: Array<() => void> = []
    const win = getCurrentWebviewWindow()

    // stream:chunk — append delta to the current assistant message
    // Use window-scoped listen() to avoid cross-window event leakage
    win.listen<StreamChunkPayload>("stream:chunk", event => {
      setMessages(prev => {
        const last = prev[prev.length - 1]
        if (last?.role === "assistant") {
          // Append to existing assistant message
          const prevText =
            last.parts
              .filter(p => p.type === "text")
              .map(p => (p as { type: "text"; text: string }).text)
              .join("") ?? ""
          const newText = prevText + event.payload.delta
          const newParts = [
            ...last.parts.filter(p => p.type !== "text"),
            { type: "text" as const, text: newText },
          ]
          return [
            ...prev.slice(0, -1),
            { ...last, content: newText, parts: newParts },
          ]
        } else {
          // Start new assistant message
          const newMsg: UIMessage = {
            id: nanoid(),
            role: "assistant",
            content: event.payload.delta,
            parts: [{ type: "text", text: event.payload.delta }],
          }
          return [...prev, newMsg]
        }
      })
    }).then(fn => unlisten.push(fn))

    // stream:tool_call — add tool invocation part (state: "calling")
    win.listen<StreamToolCallPayload>("stream:tool_call", event => {
      setMessages(prev => {
        const last = prev[prev.length - 1]
        if (last?.role !== "assistant") return prev
        const toolPart = {
          type: "tool-invocation" as const,
          toolCallId: event.payload.tool_call_id,
          toolName: event.payload.tool_name,
          args: event.payload.args,
          state: "call" as const,
        }
        return [
          ...prev.slice(0, -1),
          { ...last, parts: [...last.parts, toolPart] },
        ]
      })
    }).then(fn => unlisten.push(fn))

    // stream:tool_result — update tool invocation to state "result"
    win.listen<StreamToolResultPayload>("stream:tool_result", event => {
      setMessages(prev => {
        const last = prev[prev.length - 1]
        if (last?.role !== "assistant") return prev
        const newParts = last.parts.map(p => {
          if (
            p.type === "tool-invocation" &&
            (p as { toolCallId?: string }).toolCallId === event.payload.tool_call_id
          ) {
            return {
              ...p,
              state: "result" as const,
              result: event.payload.result,
            }
          }
          return p
        })
        return [...prev.slice(0, -1), { ...last, parts: newParts }]
      })
    }).then(fn => unlisten.push(fn))

    // stream:done — mark loading as done
    win.listen<StreamDonePayload>("stream:done", _event => {
      setIsLoading(false)
    }).then(fn => unlisten.push(fn))

    // stream:error — set error state
    win.listen<StreamErrorPayload>("stream:error", event => {
      setError(new Error(event.payload.message))
      setIsLoading(false)
    }).then(fn => unlisten.push(fn))

    // tool:approval_request — forward to caller
    win.listen<ToolApprovalRequestPayload>("tool:approval_request", event => {
      optionsRef.current.onToolApprovalRequest?.({
        toolCallId: event.payload.tool_call_id,
        toolName: event.payload.tool_name,
        args: event.payload.args,
      })
    }).then(fn => unlisten.push(fn))

    return () => {
      unlisten.forEach(fn => fn())
    }
  }, [options.chatId]) // re-register when chatId changes

  // ── append ────────────────────────────────────────────────────────────────────
  const append = useCallback(
    async (text: string) => {
      const opts = optionsRef.current
      if (!opts.chatId) return

      // Build the user message
      const userMsg: UIMessage = {
        id: nanoid(),
        role: "user",
        content: text,
        parts: [{ type: "text", text }],
      }

      // Convert current messages to ChatMessage format for Rust
      const chatMessages: ChatMessage[] = [
        ...messagesRef.current.map(m => uiMessageToChatMessage(m)),
        { id: userMsg.id, role: "user", content: text },
      ]

      lastUserMessageRef.current = { id: userMsg.id, role: "user", content: text }

      setMessages(prev => [...prev, userMsg])
      setIsLoading(true)
      setError(null)

      try {
        await invoke("start_chat_stream", {
          windowLabel,
          folder: opts.folder,
          chatId: opts.chatId,
          messages: chatMessages,
          model: opts.model,
          provider: opts.provider,
          enabledTools: opts.enabledTools ?? [],
        })
      } catch (err) {
        setError(err instanceof Error ? err : new Error(String(err)))
        setIsLoading(false)
      }
    },
    [windowLabel]
  )

  // ── stop ─────────────────────────────────────────────────────────────────────
  const stop = useCallback(() => {
    void invoke("stop_chat_stream", { windowLabel })
    setIsLoading(false)
  }, [windowLabel])

  // ── reload ────────────────────────────────────────────────────────────────────
  const reload = useCallback(async () => {
    // Re-send the last user message
    const last = lastUserMessageRef.current
    if (!last) return
    // Remove last assistant message if present
    setMessages(prev => {
      const lastMsg = prev[prev.length - 1]
      if (lastMsg?.role === "assistant") return prev.slice(0, -1)
      return prev
    })
    await append(last.content)
  }, [append])

  const clearError = useCallback(() => setError(null), [])

  return { messages, setMessages, isLoading, error, clearError, append, stop, reload }
}

// ── Conversion helpers ────────────────────────────────────────────────────────

function uiMessageToChatMessage(msg: UIMessage): ChatMessage {
  const textContent = msg.parts
    .filter(p => p.type === "text")
    .map(p => (p as { type: "text"; text: string }).text)
    .join("")

  const toolCalls = msg.parts
    .filter(p => p.type === "tool-invocation" && (p as { state?: string }).state === "call")
    .map(p => {
      const tp = p as {
        toolCallId: string
        toolName: string
        args: unknown
      }
      return { id: tp.toolCallId, name: tp.toolName, args: tp.args }
    })

  const toolResults = msg.parts
    .filter(p => p.type === "tool-invocation" && (p as { state?: string }).state === "result")
    .map(p => {
      const tp = p as {
        toolCallId: string
        result: unknown
        isError?: boolean
      }
      return {
        tool_call_id: tp.toolCallId,
        content: JSON.stringify(tp.result ?? ""),
        is_error: tp.isError ?? false,
      }
    })

  return {
    id: msg.id,
    role: msg.role,
    content: textContent,
    tool_calls: toolCalls.length > 0 ? toolCalls : undefined,
    tool_results: toolResults.length > 0 ? toolResults : undefined,
  }
}
